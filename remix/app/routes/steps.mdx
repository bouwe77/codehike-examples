```jsx src/App.js focus=4,8
import { useState } from "react"

export default function App() {
  const [counter, setCounter] = useState(0)
  return (
    <div>
      <h1>Tering Henkie</h1>
      <p>
        <button onClick={() => setCounter((counter) => counter + 1)}>
          {counter}
        </button>
      </p>
    </div>
  )
}
```
